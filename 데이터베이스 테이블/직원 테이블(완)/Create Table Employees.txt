# Employees 테이블 생성
create table employees(
company_id varchar2(25),
department_id varchar2(10),
employee_id varchar2(25),
name varchar2(30) not null,
gender varchar2(10) not null,
birthdate DATE not null,
phone_num varchar2(20) null,
email varchar2(320) null,
address number(7,0) not null,
grade_id varchar2(10),
job varchar2(100) null,
work_type_name varchar2(10),
employment_date DATE not null,
exit_flag number(1,0) default 0 not null,
primary key (company_id, employee_id),
foreign key
    (company_id, department_id)
references
    departments (company_id, department_id),
foreign key
    (grade_id)
references
    grades (grade_id));
