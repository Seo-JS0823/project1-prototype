create table departments(
company_id varchar2(25) not null,
department_id varchar2(10) not null,
department_name varchar2(60) not null,
manager_id varchar2(30) null,
primary key (company_id, department_id),
foreign key
    (company_id)
references
    business (company_id),
foreign key
    (manager_id)
references
    employees (employee_id));
