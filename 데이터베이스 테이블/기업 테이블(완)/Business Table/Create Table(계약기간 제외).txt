# 기업 정보 테이블 (계약 기간, 계약 만료일 제외 버전)
create table business(
company_name varchar2(50) not null,
kor_business_number varchar2(20) not null unique,
ceo_name varchar2(20) not null,
company_id varchar2(25) not null,
password varchar2(25) not null,
address varchar2(300) null,
industry_main varchar2(5) not null,
industry_sub varchar2(3) not null,
registration_date date default sysdate not null,
primary key
   (company_id),
foreign key
   (industry_main, industry_sub)
references
   industry_sub (major_code, sub_code)
);

company_name		Not null	varchar2(50)	
kor_business_number	Not null	varchar2(20)	UQ
ceo_name		Not null	varchar2(20)	
company_id		Not null	varchar2(25)	PK
password		Not null	varchar2(25)	
address					varchar2(300)	
industry_main		Not null	varchar2(5)	FK1
industry_sub		Not null	varchar2(3)	FK1
registration_date	Not null	date
